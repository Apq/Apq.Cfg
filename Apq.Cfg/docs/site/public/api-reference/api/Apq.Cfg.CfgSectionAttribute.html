Remarks

使用示例：

[CfgSection("Database")]
public partial class DatabaseConfig
{
    public string ConnectionString { get; set; }
    public int Timeout { get; set; } = 30;
}

// 使用生成的绑定方法
var config = DatabaseConfig.BindFrom(cfg.GetSection("Database"));
